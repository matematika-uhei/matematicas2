  1. \begin{equation*} p(X=10)=\binom{10}{10} \cdot \left( \dfrac{1}{2} \right)^{10} \cdot \left( \dfrac{1}{2} \right)^{0}= \dfrac{1}{1024} \end{equation*}
  2. \begin{equation*} p(X=3)=\binom{10}{3} \cdot \left( \dfrac{1}{2} \right)^{3} \cdot \left( \dfrac{1}{2} \right)^{7}= \dfrac{120}{1024} \end{equation*}
